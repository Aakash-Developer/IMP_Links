// range Slider Componnet ----------------------------------------------------------------------------------------------------

import React from 'react';

function RangeSlider({ name, min, max, value, onChange }) {
  return (
    <div className="w-full md:w-1/3">
      <input
        type="range"
        name={name}
        min={min}
        max={max}
        value={value}
        onChange={onChange}
        className="w-full"
      />
      <div className="text-center">{value}</div>
    </div>
  );
}

export default RangeSlider;

//------------------------------------------------------------------------------------------------------------------------------ use

import React, { useState } from 'react';
import RangeSlider from './RangeSlider'; // Adjust the path based on your file structure

function App() {
  const [skillData, setSkillData] = useState({
    level: 3, // Set an initial value
  });

  const handleLevelChange = (event) => {
    const newValue = parseInt(event.target.value);
    setSkillData({ ...skillData, level: newValue });
  };

  return (
    <div className="App">
      {/* Use the RangeSlider component */}
      <RangeSlider
        name="level"
        min={0}
        max={5}
        value={skillData.level}
        onChange={handleLevelChange}
      />
    </div>
  );
}

export default App;
